cmake_minimum_required (VERSION 3.1)
project (OHCS)
# The version number.
set (CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -fopenmp -Wall")
set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++0x -fopenmp -Wall -I/usr/lib/gcc/x86_64-redhat-linux/7/include/ -I./include -I./bigint")
set (OHCS_VERSION_MAJOR 0)
set (OHCS_VERSION_MINOR 1)
ADD_SUBDIRECTORY (lib)
ADD_SUBDIRECTORY (main)
find_package(OpenMP)
if (OPENMP_FOUND)
	set (CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
	set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
	set (CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${OpenMP_EXE_LINKER_FLAGS}")
endif()
